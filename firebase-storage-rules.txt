rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // 공개 다운로드 허용 여부를 컨트롤하세요.
    // 현재는 읽기 모두 허용, 쓰기 인증 필요 예시입니다.
    match /{allPaths=**} {
      // 읽기: 모두 허용 (필요 시 request.auth != null 로 제한)
      allow read: if true;
      // 쓰기: 로그인된 사용자만
      allow write: if request.auth != null;
    }
  }
}

## 적용 방법
1) Firebase Console → Storage → 규칙 탭
2) 위 규칙을 붙여넣고 저장 후 게시

## 권장 설정 팁
- 실제 서비스에서는 read도 인증 제한을 검토하세요:
  allow read: if request.auth != null;
- 파일 타입 제한을 추가하려면 커스텀 조건을 더하세요:
  && (request.resource.contentType.matches('image/.*'))
- 공개 다운로드가 필요한 공개 이미지 경로만 따로 허용할 수도 있습니다:
  match /public/{allPaths=**} {
    allow read: if true;
    allow write: if request.auth != null;
  }
